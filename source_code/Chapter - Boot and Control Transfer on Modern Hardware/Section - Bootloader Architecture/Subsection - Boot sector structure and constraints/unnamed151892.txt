,caption=Minimal BIOS boot sector (NASM),label={lst:bootsector}]
BITS 16
org 0x7C00

start:
    cli                     ; disable interrupts during init
    xor ax, ax
    mov ds, ax              ; set data segment to 0
    mov es, ax
    mov ss, ax
    mov sp, 0x7C00          ; simple stack at 0x0000:0x7C00 (safe region)
    sti                     ; re-enable interrupts if desired

    mov si, msg             ; DS:SI -> string
.print:
    lodsb                   ; AL = [DS:SI++]
    or al, al
    jz .hang
    mov ah, 0x0E            ; BIOS teletype function
    mov bh, 0x00
    mov bl, 0x07            ; attribute: white on black
    int 0x10
    jmp .print

.hang:
    cli
    hlt
    jmp .hang

msg db 'Boot sector: BIOS control transferred', 0

; fill until 510 bytes, then signature
times 510-($-$$) db 0
dw 0xAA55                 ; boot signature
